function(e) {
  e.target; // 유저가 실제로 누른 것
  e.currentTarget; // 이벤트리스너 달린 곳
  == this;
  e.preventDefault(); // 이벤트 기본동작 막아줌
  e.stopPropagation(); // 내 상위요소로 이벤트 버블링 막아줌
}
